FROM T_AM_PROCESSO P, T_AM_CLIENTE C 
WHERE P.CD_PESSOA1 = P.NR_PROCESSO
AND C.DS_RAZAO_SOCIAL = ?
AND P.DT_FECHAMENTO IS NULL;

SELECT P.NR_PROCESSO
FROM T_AM_PROCESSO P, T_AM_CLIENTE CL
WHERE P.CD_PESSOA1 = CL.CD_PESSOA
AND CL.DS_RAZAO_SOCIAL = 'XPTO Padaria'
AND P.DT_FECHAMENTO IS NULL;

 --NOME CLIENTE SUBSELECT
 
 
 select NR_PROCESSO  from  
  (
    select * from T_AM_CLIENTE CL ,T_AM_PROCESSO P
    where P.CD_PESSOA1 = CL.CD_PESSOA
    AND P.DT_FECHAMENTO  IS NULL 
    AND CL.DS_RAZAO_SOCIAL='Magazine luiza'
  
  
  ); 
 
 


  --PERIODO 
  
	SELECT
	 PR.NR_PROCESSO,
	 PR.DS_PROCESSO, 
	 CLI.DS_RAZAO_SOCIAL
	FROM
	 T_AM_PROCESSO PR
  INNER JOIN T_AM_CLIENTE CLI ON PR.CD_PESSOA1 = CLI.CD_PESSOA
	WHERE
    PR.DT_ABERTURA >= ? AND PR.DT_ABERTURA <= ?
	AND PR.DT_FECHAMENTO is null;
  
  
  
  --NUMERO DO PROCESSO
  
  SELECT 
    PR.NR_PROCESSO,
    PR.DS_PROCESSO,
    CLI.DS_RAZAO_SOCIAL
  FROM
    T_AM_PROCESSO PR,
    T_AM_CLIENTE CLI
  INNER JOIN T_AM_PROCESSO P ON P.NR_PROCESSO = P.NR_PROCESSO
  WHERE
  P.NR_PROCESSO = 4
  AND P.DT_FECHAMENTO IS NULL;
